CREATE TABLE games (
    id INT PRIMARY KEY,                       -- ID único de IGDB
    name VARCHAR(255) NOT NULL,
    summary TEXT,                             -- Texto largo
    slug VARCHAR(255) UNIQUE,                 -- URL amigable
    storyline TEXT,                           -- Texto largo
    cover VARCHAR(255),                       -- URL de la imagen
    genres JSONB,                             -- Array: ["RPG", "Adventure"]
    platforms JSONB,                          -- Array: ["PC", "PS5"]
    rating DECIMAL(5,2),                      -- Puntuación (86.25)
    expansions JSONB,                         -- Array de IDs: [1024, 2048]
    game_modes JSONB,                         -- Array: ["Single player", "Multiplayer"]
    url VARCHAR(255),                         -- Link a IGDB
    company VARCHAR(255),                      -- Nombre de la empresa
    fechaInsert TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE users (
    id INT PRIMARY KEY,                    -- SERIAL genera el ID automático (1, 2, 3...)
    nick VARCHAR(50) UNIQUE NOT NULL,         -- El nombre de usuario no se puede repetir
    email VARCHAR(255) UNIQUE NOT NULL,       -- El email tampoco
    password VARCHAR(255) NOT NULL,           -- Aquí guardarás el hash de la contraseña
    avatar VARCHAR(255) ,-- Ruta de la imagen
    fecha_registro TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE listas_games (
    id SERIAL PRIMARY KEY,
    user_id INT REFERENCES users(id) ON DELETE CASCADE, -- Relación con el usuario
    game_id INT REFERENCES games(id) ON DELETE CASCADE,    -- ID del juego (relacionado con la tabla juegos)
    status VARCHAR(20) CHECK (status IN ('jugando', 'completado', 'deseado', 'abandonado')),
    fecha_add TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(user_id, game_id) -- Evita que un usuario añada el mismo juego dos veces
);
